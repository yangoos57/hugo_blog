# 10 < num < 100
# 0.5 < num2 < 10
# 255 < num3 < 50

result = df.query('10 < num < 100 & 0.5 < num2 < 10 & 25< num3 < 50')

len(result)   >>>  3982

BM = (df['num'] > 10) & (df['num'] < 100)
BM2 = (df['num2'] > 0.5) & (df['num2'] < 10)
BM3 = (df['num3'] > 25) & (df['num3'] < 50)

result = df[BM & BM2 & BM3]

len(result)  >>> 3982
